                        # scss

                        $size: 100vw;

[data-filter*=image-] {
	flex: 0 1 20%;
	line-height: 0;
	position: relative;
	z-index: 0;
	

	img {
		height: auto;
		width: 100%;
	}

}

$filters: (
	grayscale: '(50%)',
	saturate: '(360%)',
	sepia: '(100%)',
	invert: '(100%)',
	opacity: '(50%)',
	brightness: '(120%)',
	contrast: '(160%)',
	hue-rotate: '(160deg)',
	blur: '(2px)'
);

@each $type, $value in $filters {
	[data-filter="image-#{$type}"] {
		img {
	      filter: #{$type}#{$value};
		}
	}
}
